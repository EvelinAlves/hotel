function ativaNoScroll() {
    elementMenu = document.querySelector('#menu');

    if(elementMenu.getBoundingClientRect().top < 0 && !elementMenu.classList.contains("fixo")) {
        elementMenu.classList.add('fixo');
        console.log("colocou");
    }
    if(elementMenu.getBoundingClientRect().top > 0) {
        elementMenu.classList.remove('fixo');
        console.log("removeu");
    }
}
